public class AddTwoNumbers {
    public static void main(String[] args) {
        int num1 = 10; // First number
        int num2 = 20; // Second number

        // Adding two numbers
        int sum = num1 + num2;

        // Displaying the result
        System.out.println("The sum of " + num1 + " and " + num2 + " is: " + sum);
    }
}
